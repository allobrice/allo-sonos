**parseSource function:**
```typescript
export function parseSource(uri: string | undefined | null): string | null {
  if (!uri) return null
  if (uri.startsWith('x-sonos-spotify:'))       return 'spotify'
  if (uri.startsWith('x-deezer-user:') ||
      uri.includes('deezer'))                    return 'deezer'
  if (uri.startsWith('x-sonosapi-radio:') ||
      uri.startsWith('x-sonosapi-stream:') ||
      uri.startsWith('x-rincon-mp3radio:') ||
      uri.startsWith('hls-radio:'))              return 'tunein'
  if (uri.startsWith('x-file-cifs:') ||
      uri.startsWith('x-rincon-playlist:'))      return 'library'
  return null  // Unknown source — do not guess
}
```

Note on Deezer: The `x-deezer-user:` prefix is MEDIUM confidence (community docs). The fallback `uri.includes('deezer')` is a safety net. Log the raw URI during development to validate on real hardware.

**StateCache class:**

Constructor takes a `broadcastFn: (event: string, data: unknown) => void` callback. This decouples the cache from the WebSocket plugin — the broadcast function is injected when the cache is created.

```typescript
export class StateCache {
  private readonly zones: Map<string, ZoneState> = new Map()
  private readonly debounceTimers: Map<string, ReturnType<typeof setTimeout>> = new Map()
  private readonly broadcastFn: (event: string, data: unknown) => void
  private static readonly DEBOUNCE_MS = 300  // User decision: ~300ms debounce

  constructor(broadcastFn: (event: string, data: unknown) => void) {
    this.broadcastFn = broadcastFn
  }

  /** Set initial state for a speaker. Used at startup before GENA events begin. */
  initialize(uuid: string, state: ZoneState): void {
    this.zones.set(uuid, state)
  }

  /** Return all zone states as an array (for snapshot on WS connect). */
  getAll(): ZoneState[] {
    return Array.from(this.zones.values())
  }

  /** Return a single zone state by UUID. */
  get(uuid: string): ZoneState | undefined {
    return this.zones.get(uuid)
  }

  /**
   * Synchronously merge partial state into the existing zone state.
   * CRITICAL: This must be synchronous — no async. The cache write
   * and scheduleUpdate call must be in the same synchronous block
   * to prevent debounce race conditions (see Pitfall 4 in research).
   */
  patch(uuid: string, partial: Partial<Omit<ZoneState, 'uuid' | 'name'>>): void {
    const existing = this.zones.get(uuid)
    if (!existing) return  // Ignore patches for unknown speakers
    Object.assign(existing, partial, { lastSeen: Date.now() })
  }

  /** Mark a speaker as reachable or unreachable. */
  markReachable(uuid: string, reachable: boolean): void {
    const existing = this.zones.get(uuid)
    if (!existing) return
    existing.reachable = reachable
    if (reachable) existing.lastSeen = Date.now()
  }

  /** Check if a speaker is currently reachable. */
  isReachable(uuid: string): boolean {
    return this.zones.get(uuid)?.reachable ?? false
  }

  /**
   * Schedule a debounced broadcast of the zone's current state.
   * Multiple rapid calls for the same UUID (e.g., volume + mute in one GENA batch)
   * result in a single broadcast after 300ms of quiet.
   */
  scheduleUpdate(uuid: string): void {
    const existing = this.debounceTimers.get(uuid)
    if (existing) clearTimeout(existing)
    this.debounceTimers.set(uuid, setTimeout(() => {
      this.debounceTimers.delete(uuid)
      const state = this.zones.get(uuid)
      if (state) {
        this.broadcastFn('state_changed', state)
      }
    }, StateCache.DEBOUNCE_MS))
  }

  /** Cancel all pending debounce timers. Call on shutdown. */
  clearTimers(): void {
    for (const timer of this.debounceTimers.values()) {
      clearTimeout(timer)
    }
    this.debounceTimers.clear()
  }
}
```

**Key design decisions:**
- `broadcastFn` is injected via constructor — no circular dependency with the websocket plugin
- `Partial<Omit<ZoneState, 'uuid' | 'name'>>` prevents accidentally overwriting identity fields
- `lastSeen` is set automatically on every patch and markReachable(true)
- `clearTimers()` is for cleanup on `fastify.close()` to prevent memory leaks
- `ReturnType<typeof setTimeout>` is used instead of `NodeJS.Timeout` for broader TS compatibility

**Step 1: Create `backend/src/plugins/websocket.ts`**

This Fastify plugin registers `@fastify/websocket` and decorates the Fastify instance with:
- `broadcast(event, data)` — sends to all connected WebSocket clients
- `stateCache` — the StateCache instance (created here, wired to broadcast)

**CRITICAL ordering rule from research:** `@fastify/websocket` MUST be registered before any routes so it can intercept upgrade requests. This plugin will be registered between the sonos plugin and routes in `app.ts`.

```typescript
import fp from 'fastify-plugin'
import type { FastifyPluginAsync } from 'fastify'
import websocketPlugin from '@fastify/websocket'
import type { WebSocket } from 'ws'
import { StateCache } from '../services/state-cache.js'

declare module 'fastify' {
  interface FastifyInstance {
    /** Broadcast a JSON message to all connected WebSocket clients */
    broadcast(event: string, data: unknown): void
    /** In-memory state cache for all Sonos zones */
    stateCache: StateCache
  }
}

const wsPlugin: FastifyPluginAsync = async (fastify) => {
  // 1. Register @fastify/websocket (must be before routes)
  await fastify.register(websocketPlugin)

  // 2. Create broadcast function
  const broadcast = (event: string, data: unknown): void => {
    const message = JSON.stringify({ event, data })
    for (const client of fastify.websocketServer.clients) {
      if ((client as WebSocket).readyState === 1 /* WebSocket.OPEN */) {
        client.send(message)
      }
    }
  }

  // 3. Create state cache, injecting broadcast function
  const stateCache = new StateCache(broadcast)

  // 4. Decorate Fastify instance
  fastify.decorate('broadcast', broadcast)
  fastify.decorate('stateCache', stateCache)

  // 5. Cleanup on shutdown — cancel debounce timers
  fastify.addHook('onClose', async () => {
    stateCache.clearTimers()
  })
}

export default fp(wsPlugin, {
  name: 'websocket-plugin',
  dependencies: ['sonos'],  // sonos plugin must run first so speakers are available
})
```

**Step 2: Create `backend/src/routes/ws.ts`**

The WebSocket route handler. On connection, synchronously push the full state snapshot before any awaits (per Pitfall 5 — event handlers must be attached synchronously).

```typescript
import type { FastifyPluginAsync } from 'fastify'

const wsRoutes: FastifyPluginAsync = async (fastify) => {
  fastify.get('/ws', { websocket: true }, (socket, _req) => {
    // SYNCHRONOUS — attach handlers and send snapshot before any async work
    // (Pitfall 5: @fastify/websocket requires synchronous handler attachment)

    // Push full state snapshot on connect
    const snapshot = fastify.stateCache.getAll()
    socket.send(JSON.stringify({ event: 'snapshot', data: snapshot }))
    fastify.log.info('[ws] Client connected, snapshot sent (%d zones)', snapshot.length)

    socket.on('close', () => {
      fastify.log.info('[ws] Client disconnected')
    })

    // No incoming message handling — server→client only (per user decision)
  })
}

export default wsRoutes
```
